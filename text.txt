        function spinWheel() {
            const available = restaurants.filter(r => !r.locked);
            if (available.length === 0) {
                alert('All options locked for this week! Click "Reset Week" to spin again.');
                return;
            }
            
            const spinner = document.getElementById('spinner');
            spinner.classList.add('spinning');
            
            setTimeout(() => {
                const picked = available[Math.floor(Math.random() * available.length)];
                picked.lastChosen = new Date().toLocaleDateString();
                picked.timesChosen++;
                picked.locked = true;
                currentWeek = picked;
                saveData();
                
                spinner.classList.remove('spinning');
                
                const resultHTML = `
                    <div style="font-size: 32px; margin-bottom: 10px;">🎉</div>
                    <div style="font-size: 28px; margin-bottom: 20px;">${picked.name}</div>
                    <div class="result-details">
                        <div><strong>🍲 Food:</strong> ${picked.food}</div>
                        <div><strong>📍 Location:</strong> ${picked.location}</div>
                        <div><strong>🏠 Address:</strong> ${picked.address}</div>
                        <div><strong>📞 Phone:</strong> ${picked.phone}</div>
                    </div>
                    <div class="result-buttons">
                        <a href="tel:${picked.phone.replace(/[^0-9+]/g, '')}" class="call-btn">📞 Call to Reserve</a>

                `;
                document.getElementById('result').innerHTML = resultHTML;
                renderUI();
            }, 2000);

            const restaurant = restaurants.find(r => r.id === id);

        function saveRestaurant() {
            const name = document.getElementById('modalName').value.trim();
            const food = document.getElementById('modalFood').value.trim();
            const location = document.getElementById('modalLocation').value;
            const address = document.getElementById('modalAddress').value.trim();
            const phone = document.getElementById('modalPhone').value.trim();
            
            if (!name || !food || !location) {
                alert('Please fill in Name, Food, and Location');
                return;
            }
            
            if (editingId) {
                const r = restaurants.find(x => x.id === editingId);
                r.name = name;
                r.food = food;
                r.location = location;
                r.address = address;
                r.phone = phone;
            } else {
                restaurants.push({
                    id: nextId++,
                    name,
                    food,
                    location,
                    address,
                    phone,
                    lastChosen: null,
                    timesChosen: 0,
                    locked: false
                });
            }
            
            saveData();
            closeAddModal();
            renderUI();
        }
